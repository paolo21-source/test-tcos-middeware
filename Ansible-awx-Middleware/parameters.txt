//PURCHASE REQUEST

 public void handlePurchaseRequest(
            String providerClientTag,
            boolean isFirstSubscription,
            String activationTemplateId,
            CloudBlueRequest request,
            Set<ResourceQuota> requestResourceQuota) {

        Map<String, Object> extraVars = new HashMap<>();
        extraVars.put(AwxLaunchRequestExtraVars.PROVIDER_CLIENT_TAG_KEY, providerClientTag);
        extraVars.put(AwxLaunchRequestExtraVars.EXTERNAL_ID_KEY, request.externalId());
        extraVars.put(AwxLaunchRequestExtraVars.REQUEST_ID_KEY, request.id());
        extraVars.put(AwxLaunchRequestExtraVars.PRODUCT_ID_KEY, request.productId());
        extraVars.put(AwxLaunchRequestExtraVars.HUB_ID_KEY, request.hubId());
        extraVars.put(AwxLaunchRequestExtraVars.USER_ID_KEY, request.userId());
        extraVars.put(AwxLaunchRequestExtraVars.USER_EMAIL_KEY, request.emailAddress());
        extraVars.put(AwxLaunchRequestExtraVars.SUBSCRIPTION_ID_KEY, request.subscriptionId());
        extraVars.put(AwxLaunchRequestExtraVars.IS_FIRST_SUBSCRIPTION_KEY, isFirstSubscription);
        extraVars.put(AwxLaunchRequestExtraVars.ACTIVATION_TEMPLATE_ID_KEY, activationTemplateId);
        extraVars.putAll(mapQuotasToExtraVars(requestResourceQuota));

        launchJobTemplate(
                awxClientProperties.getPurchaseRequestJobId(),
                AwxLaunchRequest.builder()
                        .extraVars(extraVars)
                        .build());
    }
 
private String mapQuotaToExtraVar(ResourceQuota quota) {
        return switch (quota.name()) {
            case VCPU, VCPU_PAY_AS_YOU_GO -> AwxLaunchRequestExtraVars.QUOTA_VCPU_KEY;
            case RAM, RAM_PAY_AS_YOU_GO -> AwxLaunchRequestExtraVars.QUOTA_RAM_KEY;
            case STORAGE, STORAGE_PAY_AS_YOU_GO -> AwxLaunchRequestExtraVars.QUOTA_STORAGE_KEY;
            case PUB_IP, PUB_IP_PAY_AS_YOU_GO -> AwxLaunchRequestExtraVars.QUOTA_PUB_IP_KEY;
            case BACKUP, BACKUP_PAY_AS_YOU_GO -> AwxLaunchRequestExtraVars.QUOTA_BACKUP_KEY;
            case MSWINSVR_LICENSE, MSWINSVR_LICENSE_PAY_AS_YOU_GO ->
                    AwxLaunchRequestExtraVars.QUOTA_MSWINSVR_LICENSE_KEY;
            case RHLINENT_LICENSE, RHLINENT_LICENSE_PAY_AS_YOU_GO ->
                    AwxLaunchRequestExtraVars.QUOTA_RHLINENT_LICENSE_KEY;
        };
